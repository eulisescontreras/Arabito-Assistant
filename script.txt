-- Database: "arabito-assistant"

-- DROP DATABASE "arabito-assistant";

CREATE DATABASE "arabito-assistant"
  WITH OWNER = root
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'es_VE.UTF-8'
       LC_CTYPE = 'es_VE.UTF-8'
       CONNECTION LIMIT = -1;

-- Schema: public

-- DROP SCHEMA public;

CREATE SCHEMA public
  AUTHORIZATION postgres;

GRANT ALL ON SCHEMA public TO postgres;
GRANT ALL ON SCHEMA public TO public;
COMMENT ON SCHEMA public
  IS 'standard public schema';

-- Table: public.bills

-- DROP TABLE public.bills;

CREATE TABLE public.bills
(
  id serial NOT NULL,
  amount real,
  created_at date,
  expiration_at date,
  spend character varying(100) DEFAULT 'WAITING',
  CONSTRAINT bills_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.bills
  OWNER TO root;
  
-- Table: public."workers"

-- DROP TABLE public."workers";

CREATE TABLE public."workers"
(
  id serial NOT NULL,
  name character varying(100),
  email character varying(100),
  mobile character varying(100),
  daily_s real,
  charge character varying(100),
  birth date,
  CONSTRAINT "workers_pkey" PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public."workers"
  OWNER TO root;

-- Table: public."days"

-- DROP TABLE public."days";

CREATE TABLE public."days"
(
  id serial NOT NULL,
  date date,
  worker integer,
  CONSTRAINT "days_pkey" PRIMARY KEY (id),
  CONSTRAINT "days_worker_fkey" FOREIGN KEY (worker)
      REFERENCES public."workers" (id) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public."days"
  OWNER TO root;
